-- Users Table
-- This table stores user information, which is managed by Firebase Authentication.
-- We can create a 'users' table to store additional user-specific data.
CREATE TABLE users (
    id VARCHAR(255) PRIMARY KEY, -- Corresponds to Firebase UID
    display_name VARCHAR(255),
    email VARCHAR(255) UNIQUE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Topics Table
-- Stores the topics that users can choose to focus on.
CREATE TABLE topics (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    image_path VARCHAR(255),
    color VARCHAR(50)
);

-- Courses Table
-- Stores information about meditation courses.
CREATE TABLE courses (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    subtitle VARCHAR(255),
    description TEXT,
    duration VARCHAR(50),
    image_asset VARCHAR(255),
    color VARCHAR(50)
);

-- Sessions Table
-- Stores the individual sessions within a course.
CREATE TABLE sessions (
    id INT PRIMARY KEY AUTO_INCREMENT,
    course_id INT,
    title VARCHAR(255) NOT NULL,
    duration VARCHAR(50),
    FOREIGN KEY (course_id) REFERENCES courses(id)
);

-- Music Table
-- Stores information about music tracks.
CREATE TABLE music (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    subtitle VARCHAR(255),
    image VARCHAR(255),
    color VARCHAR(50),
    duration VARCHAR(50),
    category VARCHAR(100)
);

-- SleepStories Table
-- Stores information about sleep stories.
CREATE TABLE sleep_stories (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    subtitle VARCHAR(255),
    image_asset VARCHAR(255),
    duration VARCHAR(50),
    category VARCHAR(100)
);

-- Favorites Table
-- Stores the user's favorite items (courses, music, sleep stories).
CREATE TABLE favorites (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id VARCHAR(255),
    item_id INT,
    item_type VARCHAR(50), -- 'course', 'music', 'sleep_story'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- Downloads Table
-- Stores the items downloaded by the user.
CREATE TABLE downloads (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id VARCHAR(255),
    item_id INT,
    item_type VARCHAR(50), -- 'course', 'music', 'sleep_story'
    download_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- Reminders Table
-- Stores meditation reminders set by the user.
CREATE TABLE reminders (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id VARCHAR(255),
    time TIME,
    days VARCHAR(255), -- e.g., "M,T,W,TH,F"
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- DailyThoughts Table
CREATE TABLE daily_thoughts (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    subtitle VARCHAR(255),
    date DATE
);

-- Recommendations Table
CREATE TABLE recommendations (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    subtitle VARCHAR(255),
    image_asset VARCHAR(255),
    color VARCHAR(50)
);
